-- VERIFICAR FUNCIONARIO TRANSFERIDOS
-- CONSULTA PLANILIA SEGURO 01/06/2009
SELECT RA_FILIAL, RA_MAT, RA_CC, RA_NOME, RA_CIC, RA_ADMISSA, RA_DEMISSA FROM SRA010
WHERE RA_SITFOLH <> 'D'
AND D_E_L_E_T_ <> '*'
AND RA_ADMISSA > '20090501'
ORDER BY RA_NOME

SELECT RA_FILIAL, RA_MAT, RA_CC, RA_NOME, RA_CIC, RA_ADMISSA, RA_DEMISSA FROM SRA010
WHERE RA_DEMISSA > '20090501'
AND D_E_L_E_T_ <> '*'
ORDER BY RA_NOME
