SELECT F1_FILIAL,F1_DOC,F1_SERIE,F1_FORNECE,F1_LOJA,A2_NOME,A2_CGC,F1_EST,
SUBSTRING(F1_EMISSAO,7,2)+'/'+SUBSTRING(F1_EMISSAO,5,2)+'/'+SUBSTRING(F1_EMISSAO,1,4)DATA,F1_CC,F1_USUARIO 
FROM SF1010 A, SA2010 B
WHERE A.F1_FORNECE = B.A2_COD
AND A.F1_LOJA = B.A2_LOJA
AND A.D_E_L_E_T_ <> '*'
AND B.D_E_L_E_T_ <> '*'
AND (A.F1_ESPECIE = 'NFF' OR A.F1_ESPECIE = 'NFS')
AND A.F1_EMISSAO >= '20080801'
ORDER BY F1_FORNECE
