SELECT E2_NUM,
       E2_PARCELA,
       E2_TIPO,
       E2_NATUREZ,
       E2_NOMFOR,
       SUBSTRING(E2_EMISSAO,7,2)+"/"+
       SUBSTRING(E2_EMISSAO,5,2)+"/"+
       SUBSTRING(E2_EMISSAO,1,4) EMISSAO,	
       STR(E2_VALOR,10,2)VALOR,
       E2_CC
FROM SE2010
WHERE E2_TIPO = 'PA'
AND E2_EMISSAO BETWEEN '20071001' AND '20071031'
AND D_E_L_E_T_ <> '*'   
ORDER BY E2_CC, E2_BAIXA
