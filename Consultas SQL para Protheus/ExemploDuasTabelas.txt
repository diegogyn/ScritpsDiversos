SELECT RA_MAT MATRICULA, RA_NOME NOME, RJ_DESC FUNCAO, SUBSTRING(RA_ADMISSA,7,2)+'/'+SUBSTRING(RA_ADMISSA,5,2)+'/'+SUBSTRING(RA_ADMISSA,1,4)ADMISSAO
FROM SRA010 A, SRJ010 B
WHERE RA_FILIAL = '00'
AND RA_CC = '240'
AND A.D_E_L_E_T_ <> '*'
AND B.D_E_L_E_T_ <> '*'
AND A.RA_CODFUNC = B.RJ_FUNCAO
ORDER BY RA_NOME
